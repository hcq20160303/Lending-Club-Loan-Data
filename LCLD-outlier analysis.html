def min_max_uniform(target):
    loan_data[target+'_uniform'] = 0
    mean_target = loan_data[target].mean()
    max_target = loan_data[target].max()
    min_target = loan_data[target].min()
    loan_data[target+'_uniform'] = (loan_data[target] - min_target)/(max_target - min_target)

def outlier_use_zscore(target, threshold):
    loan_data[target+'_is_outlier'] = False
    zscore = (loan_data[target] - loan_data[target].mean())/loan_data[target].std()
    loan_data[target+'_is_outlier'] = (zscore.abs() > threshold)
